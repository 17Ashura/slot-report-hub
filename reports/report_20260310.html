MODIFY BuffManager.OnTurnEnd() に ITickable 呼び出しを追加 BuffManager.cs
ターン減算の前に毎ターン効果を発動する
public void OnTurnEnd()
{
    for (int i = activeBuffs.Count - 1; i >= 0; i--)
    {
        var inst = activeBuffs[i];

        // ← 追加: 毎ターン効果を発動
        if (inst.Buff is ITickable tickable)
            tickable.Tick(status);

        inst.DecrementTurn();

        if (inst.IsExpired())
        {
            inst.Buff.OnExpired(status, this);
            bool affect = inst.Buff.AffectsProbability;
            activeBuffs.RemoveAt(i);
            if (affect) NotifyProbabilityChanged();
        }
    }
}
NEW FILE TickBuffData.cs / TickBuffRuntime.cs（新規）
正の値 = 回復、負の値 = ポイズン。1クラスで両方カバー
// TickBuffData.cs
[CreateAssetMenu(fileName = "TickBuff", menuName = "Data/Buffs/TickBuff")]
public class TickBuffData : BuffBase
{
    public int valuePerTurn; // 正 = 回復、負 = ポイズン
    public override IBuff CreateRuntime() => new TickBuffRuntime(this);
}

// TickBuffRuntime.cs
public class TickBuffRuntime : BuffRuntimeBase, ITickable
{
    private readonly TickBuffData _data;
    public TickBuffRuntime(TickBuffData data) : base(data) => _data = data;

    public void Tick(MainStatusRuntime status)
    {
        int val = _data.valuePerTurn;
        if (val >= 0)
            status.hp = Mathf.Min(status.hp + val, status.maxHp);  // 回復
        else
            status.hp = Mathf.Max(status.hp + val, 0);            // ポイズン
#if UNITY_EDITOR
        string label = val >= 0 ? "回復" : "ポイズン";
        Debug.Log($"[{label}] {status.characterName}: {val:+#;-#;0} (残HP:{status.hp})");
#endif
    }

    public override void Apply(BuffContext context, BuffInstance instance) { }
}
